Entity sandbox

Protip: use the console

Basic get/set

var e = new Entity();
e.set({ test : 123 });
console.log('none:', e.get('none'));
console.log('test:', e.get('test'));

Get multiple values

var e = new Entity();
e.set({ test : 123, hello: 'world' });
setTimeout(function() {
  e.set('test', 1234);

  // get multiple values
  console.log('values:', e.get(['test', 'hello']));

  // or metadata
  var ages = e.get(['test', 'hello'], 'modified');
  console.log('ages:', ages);
  console.log('age difference:', (ages[0] - ages[1]) + 'ms');

}, 100);

Traits

Entity.trait('test', {
  hello : function() {
    return 'world';
  }
});

var e = Entity.create({ traits : ['test'] });
e.hello(); // outputs 'world'
